3. archivo de transacciones .ldf	obligatorio

CREAR LA BASE DE DATOS

*/PRIMERA FORMA     (OBSERVACION:CUANDO OBSERVAMOS LA CLAUSURA ON PRIMARY ESTAMOS CREANDO EL ARCHIVO CON EXTENSION .MDF)*/

CREATE DATABASE COLEGIO
ON PRIMARY	(NAME=COLEGIO_DATA,
		 FILENAME="D:\TRABAJO\COLEGIO.MDF",
		 SIZE=5MB,
		 MAXSIZE=10MB,
		 FILEGROWTH=1MB)

/*SEGUNDA FORMA  OBSERVACION: CUANDO OBSERVAMOS LA CLAUSULA LOG ON ESTAMOS CREANDO EL ARCHIVO DE TRANSACIONES CON EXTENSION .LDF*/

CREATE DATABASE COLEGIO
ON PRIMARY	(NAME=COLEGIO_DATA,
		 FILENAME="D:\TRABAJO\COLEGIO.MDF",
		 SIZE=5MB,
		 MAXSIZE=10MB,
		 FILEGROWTH=1MB)

LOG ON	(NAME=COLEGIO_LOG,
		 FILENAME="D:\TRABAJO\COLEGIO.LDF",
		 SIZE=4MB,
		 MAXSIZE=12MB,
		 FILEGROWTH=10%)


/*TERCERA FORMA  OBSERVACION: CUANDO OBSERVAMOS LA CLAUSULA FILEGROUP ESTAMOS CREANDO EL ARCHIVO SECUNDARIO CON EXTENSION .NDF*/

CREATE DATABASE COLEGIO
ON PRIMARY	(NAME=COLEGIO_DATA,
		 FILENAME="D:\TRABAJO\COLEGIO.MDF",
		 SIZE=5MB,
		 MAXSIZE=10MB,
		 FILEGROWTH=1MB),

FILEGROUP DATOS2
		(NAME=COLEGIO_DATA1,
		 FILENAME="D:\TRABAJO\COLEGIO.NDF",
		 SIZE=8MB,
		 MAXSIZE=16MB,
		 FILEGROWTH=2MB)

LOG ON	(NAME=COLEGIO_LOG,
		 FILENAME="D:\TRABAJO\COLEGIO.LDF",
		 SIZE=4MB,
		 MAXSIZE=12MB,
		 FILEGROWTH=10%)
